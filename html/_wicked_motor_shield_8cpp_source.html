252 
253  if(motor_number >= 6){
254  return; // invalid motor_number, go no further
255  }
256 
257  //TODO: is this the "correct" sense of DIR_CW / DIR_CCW

298 void WickedMotorShield::setBrakeData(uint8_t motor_number, uint8_t brake_type){
299  uint8_t shift_register_value = get_shift_register_value(motor_number);
300  uint8_t * p_shift_register_value = &shift_register_value;
301  uint8_t brake_operation = OPERATION_NONE;
302  uint8_t dir_operation = OPERATION_NONE;
303 
304  if(motor_number >= 6){
305  return; // invalid motor_number, go no further
306  }
307 
308  // calculate the affect on the relevant shift register bits
309  // TODO: Merging this if/else with next if/else would make code more readable
310  if(brake_type == BRAKE_OFF){
311  brake_operation = OPERATION_CLEAR;
312  dir_operation = OPERATION_NONE;
313  }
314  else if(brake_type == BRAKE_SOFT){
315  brake_operation = OPERATION_SET;
316  dir_operation = OPERATION_CLEAR;
317 
318  }
319  else if(brake_type == BRAKE_HARD){
320  brake_operation = OPERATION_SET;
321  dir_operation = OPERATION_SET;
322  }
323 
324  // save / restore directionality
325  // we already know motor_number is a safe index into old_dir because we checked earlier
326  if(brake_type == BRAKE_OFF){
327  // when clearing the brake, restore the old_dir value
328  // when clearing the brake bit, set the direction bit according to the value in old_dir
329  if(old_dir[motor_number] == 1){
330  dir_operation = OPERATION_SET;
331  }
332  else{
333  dir_operation = OPERATION_CLEAR;
334  }
335  }
336  else if((brake_type == BRAKE_SOFT) || (brake_type == BRAKE_HARD)){
337  // when applying the brake, save the old_dir value
338  old_dir[motor_number] = get_motor_directionM(motor_number);
339  }
340 
341  switch(motor_number){
342  case M1:
343  apply_mask(p_shift_register_value, M1_BRAKE_MASK, brake_operation);
344  apply_mask(p_shift_register_value, M1_DIR_MASK, dir_operation);
345  break;
346  case M2:
347  apply_mask(p_shift_register_value, M2_BRAKE_MASK, brake_operation);
348  apply_mask(p_shift_register_value, M2_DIR_MASK, dir_operation);
349  break;
350  case M3:
351  apply_mask(p_shift_register_value, M3_BRAKE_MASK, brake_operation);
352  apply_mask(p_shift_register_value, M3_DIR_MASK, dir_operation);
353  break;
354  case M4:
355  apply_mask(p_shift_register_value, M4_BRAKE_MASK, brake_operation);
356  apply_mask(p_shift_register_value, M4_DIR_MASK, dir_operation);
357  break;
358  case M5:
359  apply_mask(p_shift_register_value, M5_BRAKE_MASK, brake_operation);
360  apply_mask(p_shift_register_value, M5_DIR_MASK, dir_operation);
361  break;
362  case M6:
363  apply_mask(p_shift_register_value, M6_BRAKE_MASK, brake_operation);
364  apply_mask(p_shift_register_value, M6_DIR_MASK, dir_operation);
365  break;
366  }
367 
368  set_shift_register_value(motor_number, shift_register_value);
369 }

428 void Wicked_Stepper::step(int16_t number_of_steps){
429  int steps_left = abs(number_of_steps); // how many steps to take
430 
431  // determine direction based on whether steps_to_mode is + or -:
432  if (number_of_steps > 0) {this->direction = 1;}
433  if (number_of_steps < 0) {this->direction = 0;}
434 
435 
436  // decrement the number of steps, moving one step each time:
437  while(steps_left > 0) {
438  // move only if the appropriate delay has passed:
439  if (millis() - this->last_step_time >= this->step_delay) {
440  // get the timeStamp of when you stepped:
441  this->last_step_time = millis();
442  // increment or decrement the step number,
443  // depending on direction:
444  if (this->direction == 1) {
445  this->step_number++;
446  if (this->step_number == this->number_of_steps) {
447  this->step_number = 0;
448  }
449  }
450  else {
451  if (this->step_number == 0) {
452  this->step_number = this->number_of_steps;
453  }
454  this->step_number--;
455  }
456  // decrement the steps left:
457  steps_left--;
458  // step the motor to step number 0, 1, 2, or 3:
459  stepMotor(this->step_number % 4);
460  }
461  }
462 }
